MainActivity.java

// Full Java code with all attribute usage, line by line
package com.example.myapp;
import android.content.Intent;
import android.os.Bundle;
import android.widget.*;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
EditText edtName;
Spinner spCourse;
CheckBox chkJava, chkPython;
RadioGroup radioGroup;
ToggleButton toggleBtn;
Switch switch1;
ProgressBar progressBar;
Button btnSubmit, btnNext;
TextView txtResult;

protected void onCreate(Bundle savedInstanceState){
super.onCreate(savedInstanceState);
setContentView(R.layout.activity_main);

edtName = findViewById(R.id.edtName);
spCourse = findViewById(R.id.spCourse);
chkJava = findViewById(R.id.chkJava);
chkPython = findViewById(R.id.chkPython);
radioGroup = findViewById(R.id.radioGroup);
toggleBtn = findViewById(R.id.toggleBtn);
switch1 = findViewById(R.id.switch1);
progressBar = findViewById(R.id.progressBar);
btnSubmit = findViewById(R.id.btnSubmit);
btnNext = findViewById(R.id.btnNext);
txtResult = findViewById(R.id.txtResult);

String[] courses = {"CS","IT","Business"};
ArrayAdapter adapter = new ArrayAdapter(this, android.R.layout.simple_spinner_item, courses);
spCourse.setAdapter(adapter);

btnSubmit.setOnClickListener(v -> {
String name = edtName.getText().toString();
String course = spCourse.getSelectedItem().toString();
String skills = "";
if(chkJava.isChecked()) skills += "Java ";
if(chkPython.isChecked()) skills += "Python ";
int id = radioGroup.getCheckedRadioButtonId();
RadioButton rb = findViewById(id);
String gender = rb.getText().toString();
String toggle = toggleBtn.isChecked() ? "ON" : "OFF";
String sw = switch1.isChecked() ? "Enabled" : "Disabled";
progressBar.setProgress(100);
String result = "Name: " + name + "\nCourse: " + course + "\nGender: " + gender + "\nSkills: " + skills + "\nToggle: " + toggle + "\nSwitch: " + sw;
txtResult.setText(result);
Toast.makeText(this, result, Toast.LENGTH_LONG).show();
Intent i = new Intent(MainActivity.this, SecondActivity.class);
i.putExtra("data", result);
startActivity(i);
});
}

package com.example.myapp;

txtData = findViewById(R.id.txtData);
